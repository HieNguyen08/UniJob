  (1) Sinh viên -- bên nhận task (tìm thu nhập, tích lũy kinh nghiệm) \newline

\textbf{Value Propositions} &
  Kết nối task nội bộ có cấu trúc; xác thực danh tính qua email trường;
  hệ thống đánh giá và xây dựng uy tín; CV Passport số hóa kinh nghiệm \\
\hline
\textbf{Channels} &
  Website (Firebase Hosting); email trường; fanpage campus; word-of-mouth \\
\hline
\textbf{Customer Relationships} &
  Tự phục vụ (self-service) qua nền tảng; hệ thống thông báo push/email \\
\hline
\textbf{Revenue Streams} &
  Transaction fee (3--5\%); gói subscription tổ chức; premium listing; campus advertising \\
\hline
\textbf{Key Resources} &
  Nền tảng web (React + Firebase); dữ liệu người dùng; hệ thống uy tín \\
\hline
\textbf{Key Activities} &
  Phát triển và vận hành nền tảng; xác thực người dùng; quản lý chất lượng task \\
\hline
\textbf{Key Partnerships} &
  Phòng CTSV; Đoàn--Hội trường; CLB kỹ thuật; Firebase (Google Cloud) \\
\hline
\textbf{Cost Structure} &
  Firebase hosting \& services ($\sim$500k VNĐ/tháng); marketing activation;
  chi phí vận hành nội dung ($\sim$2,5 triệu VNĐ/tháng tổng) \\
\hline
