\subsection{Chiến lược theo từng giai đoạn khách hàng}

\subsubsection{Giai đoạn 1 -- Awareness \& Acquisition (tháng 1--3)}

\textbf{Mục tiêu:} Đạt 1.000 tài khoản đăng ký; 50 task/tháng; tỷ lệ hoàn thành $\geq 70\%$.

\textbf{Marketing actions:}
\begin{itemize}
  \item Booth giới thiệu tại hành lang khoa KHMT, thư viện (chi phí: 2.000.000 VNĐ)
  \item Poster/social media asset thiết kế nội bộ (chi phí: 1.000.000 VNĐ)
  \item Quà tặng kích hoạt người dùng đầu tiên -- sticker, voucher (chi phí: 1.500.000 VNĐ)
  \item Hợp tác với BTV truyền thông các CLB để đăng bài giới thiệu (miễn phí)
\end{itemize}

\subsection{Mục tiêu và lợi nhuận kỳ vọng}

\begin{itemize}
  \item \textbf{Năm 1:} Doanh thu 52--68 triệu VNĐ (kịch bản cơ sở); chi phí $\approx$ 30 triệu
    $\Rightarrow$ lợi nhuận $\approx$ 22--38 triệu VNĐ.
  \item \textbf{Kỳ vọng người dùng:} 100 tài khoản/tháng từ tháng 2 trở đi (dựa trên
    2 CLB lớn trong trường $\times$ 50 thành viên mỗi CLB).
  \item \textbf{Kỳ vọng task:} 50 task/tháng vào tháng 3 $\rightarrow$ 180 task/tháng vào
    tháng 12 (tăng trưởng theo học kỳ mới).
\end{itemize}
